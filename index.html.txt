let CQ=null, allSessions=[];
const QR_MAP={'excellent':5,'good':4,'satisfactory':3,'poor':2,'unsatisfactory':1};

  const f=document.getElementById('fileInput').files;
  document.getElementById('fileList').textContent=f.length?`${f.length} fichier(s) : ${[...f].map(x=>x.name).join(' · ')}` :'';

async function loadFiles(){
  const files=document.getElementById('fileInput').files;
  const status=document.getElementById('authStatus');
  const btn=document.getElementById('connectBtn');
  if(!files.length){status.textContent='⚠ Sélectionne au moins un fichier';status.className='auth-status err';return;}
  btn.disabled=true;
  status.innerHTML='<span class="spinner"></span> Lecture en cours…';
  status.className='auth-status loading';
  allSessions=[];
  for(const file of files){
    try{allSessions.push(...await readExcel(file));}
    catch(e){status.textContent=`✗ Erreur ${file.name}: ${e.message}`;status.className='auth-status err';btn.disabled=false;return;}
  }
  if(!allSessions.length){status.textContent='✗ Aucune donnée valide.';status.className='auth-status err';btn.disabled=false;return;}
  const dates=allSessions.map(r=>r.date).filter(Boolean).sort((a,b)=>b-a);
  if(dates.length){const d=dates[0];CQ={year:d.getFullYear(),q:Math.ceil((d.getMonth()+1)/3)};}
  else{const n=new Date();CQ={year:n.getFullYear(),q:Math.ceil((n.getMonth()+1)/3)};}
  status.textContent=`✓ ${allSessions.length} sessions chargées`;
  status.className='auth-status ok';
  document.getElementById('connBadge').textContent=`● ${allSessions.length} sessions`;
  document.getElementById('connBadge').className='conn-badge ok';
  setTimeout(()=>{
    document.getElementById('setupScreen').classList.add('hidden');
    document.getElementById('toolbar').classList.remove('hidden');
    document.getElementById('mainContent').classList.remove('hidden');
    updateQLabel();render();
  },500);
}

function readExcel(file){
  return new Promise((resolve,reject)=>{
    const reader=new FileReader();
    reader.onload=e=>{
      try{
        const wb=XLSX.read(e.target.result,{type:'array',cellDates:true});
        const ws=wb.Sheets[wb.SheetNames[0]];
        let raw=XLSX.utils.sheet_to_json(ws,{range:1,raw:false,defval:''});
        if(raw.length&&!raw[0].hasOwnProperty('Company')){
          const r0=XLSX.utils.sheet_to_json(ws,{range:0,raw:false,defval:''});
          if(r0.length&&r0[0].hasOwnProperty('Company')) raw=r0;
        }
        const rows=raw.map(r=>({
          client: c(r['Company']||''),
          fstd:   c(r['Device']||''),
          date:   pd(r['Start Date & Time']||r['Start Date']||''),
          score:  pq(r['Qr']||r['QR']||''),
          hours:  pdur(r['Duration']||'1'),
        })).filter(r=>r.client&&r.client!=='Simaero'&&r.fstd&&r.date);
        resolve(rows);
      }catch(err){reject(err);}
    };
    reader.onerror=()=>reject(new Error('Lecture impossible'));
    reader.readAsArrayBuffer(file);
  });
}

function c(v){return String(v||'').trim();}
function pd(v){
  if(!v||v==='') return null;
  if(v instanceof Date&&!isNaN(v)) return v;
  const s=String(v).trim();
  const m=s.match(/^(\d{1,2})[\/\-\.](\d{1,2})[\/\-\.](\d{4})/);
  if(m) return new Date(+m[3],+m[2]-1,+m[1]);
  const m2=s.match(/^(\d{4})[\/\-\.](\d{1,2})[\/\-\.](\d{1,2})/);
  if(m2) return new Date(+m2[1],+m2[2]-1,+m2[3]);
  const d=new Date(s);if(!isNaN(d))return d;
  return null;
}
function pq(v){
  if(!v||v==='') return null;
  const s=String(v).trim().toLowerCase();
  if(QR_MAP[s]!==undefined) return QR_MAP[s];
  const n=parseFloat(s);return isNaN(n)?null:Math.min(5,Math.max(1,n));
}
function pdur(v){
  const s=String(v||'').trim();
  if(s.includes(':')){ const[h,m]=s.split(':').map(Number);return(h||0)+(m||0)/60;}
  const n=parseFloat(s);return isNaN(n)?1:n;
}

function updateQLabel(){document.getElementById('qLabel').textContent=`Q${CQ.q} ${CQ.year}`;}
function changeQ(d){CQ.q+=d;if(CQ.q>4){CQ.q=1;CQ.year++;}if(CQ.q<1){CQ.q=4;CQ.year--;}updateQLabel();render();}
function reImport(){document.getElementById('setupScreen').classList.remove('hidden');document.getElementById('connectBtn').disabled=false;document.getElementById('authStatus').textContent='';}
function qBounds(y,q){const s=[[0,1],[3,1],[6,1],[9,1]],e=[[2,31],[5,30],[8,30],[11,31]];return{start:new Date(y,s[q-1][0],s[q-1][1]),end:new Date(y,e[q-1][0],e[q-1][1],23,59,59)};}
function prevQ(y,q){return q===1?{year:y-1,q:4}:{year:y,q:q-1};}

function render(){
  const{year,q}=CQ,pq2=prevQ(year,q);
  const{start,end}=qBounds(year,q);
  const{start:ps,end:pe}=qBounds(pq2.year,pq2.q);
  const cur=allSessions.filter(r=>r.date>=start&&r.date<=end);
  const prev=allSessions.filter(r=>r.date>=ps&&r.date<=pe);
  const cm={};
  for(const r of cur){
    if(!cm[r.client])cm[r.client]={sessions:0,hours:0,fstds:{}};
    cm[r.client].sessions++;cm[r.client].hours+=r.hours;
    if(!cm[r.client].fstds[r.fstd])cm[r.client].fstds[r.fstd]={sessions:0,hours:0,scores:[]};
    cm[r.client].fstds[r.fstd].sessions++;cm[r.client].fstds[r.fstd].hours+=r.hours;
    if(r.score!==null)cm[r.client].fstds[r.fstd].scores.push(r.score);
  }
  const pm={};
  for(const r of prev){
    if(!pm[r.client])pm[r.client]={fstds:{}};
    if(!pm[r.client].fstds[r.fstd])pm[r.client].fstds[r.fstd]={scores:[]};
    if(r.score!==null)pm[r.client].fstds[r.fstd].scores.push(r.score);
  }
  const sorted=Object.entries(cm).sort((a,b)=>b[1].hours-a[1].hours);
  const allScores=cur.map(r=>r.score).filter(s=>s!==null);
  const globalAvg=allScores.length?allScores.reduce((a,b)=>a+b,0)/allScores.length:null;
  document.getElementById('summaryBar').innerHTML=`
    <div class="stat-card"><div class="val">Q${q} ${year}</div><div class="lbl">Trimestre</div></div>
    <div class="stat-card"><div class="val">${sorted.length}</div><div class="lbl">Clients actifs</div></div>
    <div class="stat-card"><div class="val">${[...new Set(cur.map(r=>r.fstd))].length}</div><div class="lbl">FSTD utilisés</div></div>
    <div class="stat-card"><div class="val">${cur.length}</div><div class="lbl">Sessions</div></div>
    <div class="stat-card"><div class="val">${Math.round(cur.reduce((s,r)=>s+r.hours,0))}h</div><div class="lbl">Heures vol</div></div>
    <div class="stat-card"><div class="val">${globalAvg!==null?globalAvg.toFixed(2)+'/5':'—'}</div><div class="lbl">Rating moyen</div></div>`;
  renderGauge(globalAvg);
  if(!sorted.length){document.getElementById('clientList').innerHTML='<div class="no-data">Aucune session pour ce trimestre.</div>';return;}
  document.getElementById('clientList').innerHTML=sorted.map(([client,d],i)=>{
    const fk=Object.keys(d.fstds);
    const sc=avg(fk.flatMap(f=>d.fstds[f].scores));
    const psc=avg(Object.values(pm[client]?.fstds||{}).flatMap(f=>f.scores));
    const fstdHtml=fk.map(fstd=>{
      const fd=d.fstds[fstd];
      const fs=avg(fd.scores);
      const pfs=avg(pm[client]?.fstds[fstd]?.scores||[]);
      return`<div class="fstd-row">
        <div class="fstd-name">✈ ${fstd}</div>
        <div class="fstd-stars">${sh(fs,'fstd-star')}</div>
        <div class="fstd-score">${fs!==null?fs.toFixed(2)+'/5':'—'}</div>
        <div class="fstd-trend ${tc(fs,pfs)}" title="vs Q${pq2.q} ${pq2.year}">${ti(fs,pfs)}</div>
        <div class="fstd-sessions">${fd.sessions} sess. · ${Math.round(fd.hours)}h</div>
      </div>`;
    }).join('');
    const delta=sc!==null&&psc!==null?(sc-psc).toFixed(2):null;
    return`<div class="client-card">
      <div class="client-header" onclick="tog(this)">
        <div class="rank-badge">${i+1}</div>
        <div class="client-name">${client}</div>
        <div class="trend-block">
          <div class="trend-score">${sc!==null?sc.toFixed(2)+'/5':'—'}</div>
          <div class="trend-stars">${sh(sc,'star')}</div>
          <div class="trend-delta ${tc(sc,psc)}">${ti(sc,psc)}${delta!==null?` <span style="font-size:11px;font-weight:400">${delta>0?'+':''}${delta}</span>`:''}</div>
        </div>
        <div class="vol-info">${d.sessions} sess. · ${Math.round(d.hours)}h</div>
        <div class="chevron">▶</div>
      </div>
      <div class="fstd-list hidden">${fstdHtml}</div>
    </div>`;
  }).join('');
}

function avg(arr){return arr&&arr.length?arr.reduce((a,b)=>a+b,0)/arr.length:null;}
